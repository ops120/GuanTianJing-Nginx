### 2. 配置解析器 (ConfigParser)

**职责**: 解析 Nginx 配置文件，提取关键信息

**解析策略**:
1. 使用正则表达式提取各类型块
2. 识别链路关系类型
3. 构建解析结果数据

**核心正则表达式**:
```javascript
// Upstream 块解析 (支持多行和单行)
/upstream\s+([\w_]+)(?::(\d+))?\s*\{((?:[^{}]|\{[^{}]*\})*)\}/g

// Server 块解析 (支持嵌套 location)
/server\s*\{((?:[^{}]|\{[^{}]*\})*)\}/g

// Location 块解析
/location\s+([^\s{]+)\s*\{([\s\S]*?)\n\s*\}/g

// Proxy Pass 解析
/proxy_pass\s+([^;]+);/

// Root 路径解析
/root\s+([^;]+);/

// IP 地址提取
/https?:\/\/(\d+\.\d+\.\d+\.\d+):?(\d+)?/

// Weight 参数提取
/weight\s*=\s*(\d+)/
```

**解析结果结构**:
```javascript
interface ParsedConfig {
    upstreams: Array<Upstream>
    servers: Array<Server>
    locations: Array<Location>
    proxies: Array<Proxy>
    links: Array<Link>
    errors: Array<Error>
}

interface Upstream {
    name: string
    port?: string
    servers: Array<string>
    source: string
    sourceFile: string
    originalName: string
}

interface Server {
    name: string
    port: string
    source: string
    sourceFile: string
    displayName?: string
}

interface Link {
    type: 'nginx_chain' | 'server_to_upstream' | 'upstream_backend' | 'server_to_static'
    source: string
    target: string
    port?: string
    location?: string
    root?: string
    sourceFile: string
    weight?: number
}
```

**链路关系提取逻辑**:
```javascript
// 1. 解析所有 upstream 块
// 2. 解析所有 server 块
// 3. 在 server 块中提取 location 块
// 4. 在 location 块中提取 proxy_pass 和 root
// 5. 根据提取内容生成链路关系
//    - proxy_pass → server_to_upstream / upstream_backend
//    - root → server_to_static
//    - server_name → IP → nginx_chain
```

### 3. 拓扑生成器 (TopologyGenerator)

**职责**: 根据链路关系构建拓扑数据结构

**布局算法**:
```javascript
function renderTopologyFromLinks(centerX: number) {
    // 1. 创建节点映射
    const nodeMap = new Map<string, Node>()
    const edgeList = []
    
    // 2. 分层布局
    // Layer 0: Client (y=60)
    // Layer 1: 第一层 Servers (y=150)
    // Layer 2: 第二层 Servers + Upstreams (y=280)
    // Layer 3: Backends (y=410)
    
    // 3. 从 links 中提取所有节点
    parsedData.links.forEach(link => {
        // 根据 link.type 创建 source 和 target 节点
        // 确保节点在 nodeMap 中存在
    })
    
    // 4. 计算节点位置
    nodeMap.forEach(node => {
        node.x = centerX - totalWidth/2 + index * spacing
        node.y = layerY
    })
    
    // 5. 构建边
    parsedData.links.forEach(link => {
        const edge = createEdge(link)
        edgeList.push(edge)
    })
}
```

**节点类型**:

enum NodeType {

### 4. SVG 渲染引擎 (Renderer)

**职责**: 将拓扑数据渲染为可交互的 SVG 图形

**渲染流程**:
```javascript
function renderSVG(skipViewBoxUpdate = false) {
    // 1. 计算 viewBox
    if (!skipViewBoxUpdate) {
        calculateViewBox()
    }
    
    // 2. 渲染渐变定义
    const gradientDefsHTML = renderGradientDefs()
    
    // 3. 渲染边（线条）
    const edgesHtml = topologyEdges.map(renderEdge)
    
    // 4. 渲染节点（圆形 + 图标）
    const nodesHtml = topologyNodes.map(renderNode)
    
    // 5. 组合输出
    svg.innerHTML = gradientDefsHTML + `<g>${edgesHtml}${nodesHtml}</g>`
}
```

**SVG 图形元素**:
```html
<svg viewBox="0 0 1000 700">
    <defs>
        <!-- 渐变定义 -->
        <linearGradient id="edgeCyan">...</linearGradient>
        <marker id="arrowCyan">...</marker>
    </defs>
    
    <!-- 边 -->
    <path d="M500,60 L500,150" stroke="url(#edgeGreen)" />
    <path class="edge-path" d="M500,60 L500,150" stroke="..." />
    
    <!-- 节点组 -->
    <g class="topology-node node" transform="translate(500, 60)">
        <circle r="25" fill="#1a1a25" stroke="#ffffff" />
        <text>Client</text>
    </g>
</svg>
```

**边渲染逻辑**:
```javascript
function renderEdge(edge, i) {
    const src = topologyNodes.find(n => n.id === edge.source)
    const tgt = topologyNodes.find(n => n.id === edge.target)
    
    // 计算路径长度和中心点
    const dx = tgt.x - src.x
    const dy = tgt.y - src.y
    const length = Math.sqrt(dx * dx + dy * dy)
    const midX = (src.x + tgt.x) / 2
    const midY = (src.y + tgt.y) / 2
    
    // 边的颜色
    const edgeColor = gradientMap[edge.type]
    
    // 链路权重（默认 1）
    const weight = edge.weight || 1
    const strokeWidth = 2 + (weight - 1) * 0.5
    
    // 路由标签处理
    let routeLabel = edge.location || edge.root
    if (routeLabel) {
        routeLabel = routeLabel.replace(/^\/+|\/+$/g, '').toUpperCase()
    }
    
    // 生成边的 SVG 元素
    return `
        <path d="M${src.x},${src.y} L${tgt.x},${tgt.y}" ... />
        <path class="edge-path" d="M${src.x},${src.y} L${tgt.x},${tgt.y}" ... />
        ${routeLabel ? `<text>${routeLabel}</text>` : ''}
        <path id="animate_${i}" d="M${src.x},${src.y} L${tgt.x},${tgt.y}" fill="none" stroke="none"/>
    `
}
```

**节点渲染逻辑**:
```javascript
function renderNode(node) {
    const colors = {
        client: { fill: '#1a1a25', stroke: '#ffffff', text: '#ffffff' },
        nginx: { fill: '#1a1a25', stroke: '#8b5cf6', text: '#8b5cf6' },
        server: { fill: '#1a1a25', stroke: '#ff00aa', text: '#ff00aa' },
        upstream: { fill: '#1a1a25', stroke: '#00d4ff', text: '#00d4ff' },
        backend: { fill: '#1a1a25', stroke: '#ffdd00', text: '#ffdd00' },
        static: { fill: '#1a1a25', stroke: '#00ff88', text: '#00ff88' }
    }
    
    const color = colors[node.type] || colors.client
    const size = node.type === 'nginx' || node.type === 'client' ? 50 : 40
    
    return `
        <g class="topology-node node" transform="translate(${node.x}, ${node.y})">
            <circle r="${size / 2}" fill="${color.fill}" stroke="${color.stroke}" />
            ${getNodeIcon(node.type, size, color)}
            <text>${node.label}</text>
        </g>
    `
}
```

### 5. 交互控制器 (InteractionController)

**职责**: 处理用户交互操作

#### 5.1 节点拖动
```javascript
// 拖动状态
let isDraggingNode = false
let draggedNodeId = null
let dragStartPos = { x: 0, y: 0 }
let dragNodeStartPos = { x: 0, y: 0 }

// 开始拖动
function startNodeDrag(event, nodeId) {
    isDraggingNode = true
    draggedNodeId = nodeId
    dragStartPos = { x: event.clientX, y: event.clientY }
    dragNodeStartPos = { x: node.x, y: node.y }
}

// 拖动中
function onNodeDrag(event) {
    const dx = event.clientX - dragStartPos.x
    const dy = event.clientY - dragStartPos.y
    
    const scaleX = viewBoxState.width / canvasWidth
    const scaleY = viewBoxState.height / canvasHeight
    
    node.x = dragNodeStartPos.x + dx * scaleX
    node.y = dragNodeStartPos.y + dy * scaleY
    
    // 跳过 viewBox 更新，防止缩放变化
    renderSVG(true)
}

// 结束拖动
function stopNodeDrag() {
    isDraggingNode = false
    draggedNodeId = null
}
```

#### 5.2 画布平移
```javascript
// 平移状态
let isPanning = false
let startPan = { x: 0, y: 0 }

// 鼠标按下开始平移
canvas.addEventListener('mousedown', (e) => {
    if (e.button === 0) {
        isPanning = true
        startPan = { x: e.clientX, y: e.clientY }
    }
})

// 鼠标移动平移
window.addEventListener('mousemove', (e) => {
    if (!isPanning) return
    
    const dx = e.clientX - startPan.x
    const dy = e.clientY - startPan.y
    
    const scaleX = viewBoxState.width / canvasWidth
    const scaleY = viewBoxState.height / canvasHeight
    
    viewBoxState.x -= dx * scaleX
    viewBoxState.y -= dy * scaleY
    
    applyPanAndZoom()
})

// 鼠标松开结束平移
window.addEventListener('mouseup', () => {
    isPanning = false
})
```

#### 5.3 滚轮缩放
```javascript
canvas.addEventListener('wheel', (e) => {
    e.preventDefault()
    
    const delta = e.deltaY > 0 ? 1.1 : 0.9
    
    // 计算鼠标在 SVG 中的位置
    const rect = canvas.getBoundingClientRect()
    const mouseX = e.clientX - rect.left
    const mouseY = e.clientY - rect.top
    
    const svgPointX = viewBoxState.x + (mouseX/rect.width) * viewBoxState.width
    const svgPointY = viewBoxState.y + (mouseY/rect.height) * viewBoxState.height
    
    // 调整 viewBox 以鼠标位置为中心缩放
    const newWidth = viewBoxState.width * delta
    const newHeight = viewBoxState.height * delta
    
    viewBoxState.x = svgPointX - (mouseX/rect.width) * newWidth
    viewBoxState.y = svgPointY - (mouseY/rect.height) * newHeight
    viewBoxState.width = newWidth
    viewBoxState.height = newHeight
    
    applyPanAndZoom()
}, { passive: false })
```

#### 5.4 Tooltip 位置计算
```javascript
function showTooltip(event, nodeId) {
    const node = topologyNodes.find(n => n.id === nodeId)
    if (!node) return
    
    // 使用节点坐标而不是鼠标坐标
    let left = node.x + padding
    let top = node.y + padding
    
    // 检查右边界
    if (left + tooltipWidth > window.innerWidth - padding) {
        left = node.x - tooltipWidth - padding
    }
    
    // 检查下边界
    if (top + tooltipHeight > window.innerHeight - padding) {
        top = node.y - tooltipHeight - padding
    }
    
    tooltip.style.left = left + 'px'
    tooltip.style.top = top + 'px'
}
```
